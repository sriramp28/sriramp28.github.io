% A wrapping, page-break-friendly two-column resume line.
% Unlike tabular*/minipage-based resume templates, this keeps the long text
% in ordinary paragraph mode, so TeX can wrap it naturally and page breaks
% can occur between list items without manual hacks.
\newlength{\resumeDateWidth}
\setlength{\resumeDateWidth}{0.5in}

\newlength{\resumeDateGap}
\setlength{\resumeDateGap}{0.12in}

\newcommand{\resumeLineWithDate}[2]{%
  \par\noindent
  \begingroup
    \rightskip=\dimexpr\resumeDateWidth+\resumeDateGap\relax
    \parfillskip=-\rightskip
    #1%
    \if\relax\detokenize{#2}\relax
    \else
      \hfill\makebox[\dimexpr\resumeDateWidth+\resumeDateGap\relax][r]{#2}%
    \fi
    \par
  \endgroup
}

\newcommand{\resumeWrappedLine}[1]{%
  \par\noindent
  \begingroup
    \rightskip=\dimexpr\resumeDateWidth+\resumeDateGap\relax
    #1\par
  \endgroup
}

\newcommand{\resumeSubheading}[4]{%
  \vspace{1pt}\item
  \begingroup
    \setlength{\parindent}{0pt}%
    \setlength{\parskip}{0pt}%
    \small
    \resumeLineWithDate{{\normalsize\textbf{#1}}}{#2}%
    \vspace{2pt}%
    \resumeWrappedLine{{\emph{#3}}}%
    \if\relax\detokenize{#4}\relax
    \else
      \resumeLineWithDate{}{#4}%
    \fi
  \endgroup
  \vspace{1pt}%
}

  \endgroup
  \vspace{2pt}%
}
